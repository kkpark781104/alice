-- 사용자 테이블
CREATE TABLE `MEMBER` (
	`id` INT(11) NOT NULL AUTO_INCREMENT COMMENT '기본키',
	`userId` VARCHAR(200) NULL DEFAULT NULL COMMENT '사용자Id',
	`userPass` VARCHAR(200) NULL DEFAULT NULL COMMENT '사용자비번',
	`userName` VARCHAR(200) NULL DEFAULT NULL COMMENT '사용자명',
	`userEmail` VARCHAR(200) NULL DEFAULT NULL COMMENT '사용자이메일',
	PRIMARY KEY (`id`)
)
COLLATE='utf8_general_ci'
ENGINE=InnoDB
AUTO_INCREMENT=3
;

-- 메뉴 테이블
CREATE TABLE `MENU` (
	`id` INT(11) NOT NULL AUTO_INCREMENT COMMENT '기본키',
	`menuName` VARCHAR(200) NULL DEFAULT NULL COMMENT '메뉴명',
	`menuUrl` VARCHAR(200) NULL DEFAULT NULL COMMENT '메뉴url',
	`menuContent` VARCHAR(200) NULL DEFAULT NULL COMMENT '메뉴내용',
	`menuSort` VARCHAR(200) NULL DEFAULT NULL COMMENT '메뉴순서',
	`isUse` VARCHAR(200) NULL DEFAULT NULL COMMENT '사용여부',
	`createDt` VARCHAR(200) NULL DEFAULT NULL COMMENT '생성일',
	`createMemberId` VARCHAR(200) NULL DEFAULT NULL COMMENT '생성자Id',
	`modifyDt` VARCHAR(200) NULL DEFAULT NULL COMMENT '수정일',
	`modifyUserId` VARCHAR(200) NULL DEFAULT NULL COMMENT '수정자Id',
	PRIMARY KEY (`id`)
)
COLLATE='utf8_general_ci'
ENGINE=InnoDB
AUTO_INCREMENT=3
;
